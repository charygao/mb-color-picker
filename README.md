## Install

```bash
npm install --save mb-color-picker
```

## Usage

```jsx
import React, { Component } from 'react'

import ColorPicker from 'mb-color-picker'

class Example extends ColorPicker {
  render () {
    return (
      <ColorPicker />
    )
  }
}
```
